📊 数据库操作

【连接 MongoDB】
docker exec -it cve-mongodb mongosh -u admin -p secure_password cve_database

【查询 CVE 总数】
docker exec cve-mongodb mongosh -u admin -p secure_password --authenticationDatabase admin cve_database --eval "db.cve_collection.countDocuments({})" --quiet

【查询 Dell 总数】
docker exec cve-mongodb mongosh -u admin -p secure_password --authenticationDatabase admin cve_database --eval "db.dell_collection.countDocuments({})" --quiet

【连接 Redis】
docker exec -it cve-redis redis-cli -a defaultpassword

【查看 Redis 缓存数量】
docker exec cve-redis redis-cli -a defaultpassword DBSIZE

【清空 Redis 缓存】
docker exec cve-redis redis-cli -a defaultpassword FLUSHDB
